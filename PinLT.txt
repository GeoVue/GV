			function GDdestinationPoint(lat, lon, bearingDeg, distanceMeters) {
				const R = 6371000, phi1 = GDdegToRad(lat), lambda1 = GDdegToRad(lon), theta = GDdegToRad(bearingDeg), delta = distanceMeters / R;
				const phi2 = Math.asin(Math.sin(phi1) * Math.cos(delta) + Math.cos(phi1) * Math.sin(delta) * Math.cos(theta));
				const lambda2 = lambda1 + Math.atan2(Math.sin(theta) * Math.sin(delta) * Math.cos(phi1), Math.cos(delta) - Math.sin(phi1) * Math.sin(phi2));
				return { lat: GDradToDeg(phi2), lon: GDradToDeg(lambda2) };
			}

			function GDgenerateCrossPoints(lat, lon, radiusMeters) {
				return { N: GDdestinationPoint(lat, lon, 0, radiusMeters), E: GDdestinationPoint(lat, lon, 90, radiusMeters), S: GDdestinationPoint(lat, lon, 180, radiusMeters), W: GDdestinationPoint(lat, lon, 270, radiusMeters) };
			}

			function GDfiniteDifferenceGradient(zN, zE, zS, zW, d) {
				const gx = (zE - zW) / (2 * d), gy = (zN - zS) / (2 * d);
				return { gx, gy, magnitude: Math.sqrt(gx * gx + gy * gy), direction: (GDradToDeg(Math.atan2(-gx, -gy)) + 360) % 360 };
			}

				});
			}

			async function GDfetchGoogleElevationsChunk(latLngLocations, attempt = 1) {
				const elevator = new google.maps.ElevationService();
				return new Promise((resolve, reject) => {
					elevator.getElevationForLocations({ locations: latLngLocations }, async function(results, status) {
						if (status === 'OK' && results && results.length === latLngLocations.length) { resolve(results.map(result => result.elevation)); return; }
						if (status === 'OVER_QUERY_LIMIT' && attempt < 4) { await GDsleep(700 * attempt); try { resolve(await GDfetchGoogleElevationsChunk(latLngLocations, attempt + 1)); } catch (err) { reject(err); } return; }
						reject(new Error('Google Elevation service failed: ' + status));
					});
				});
			}

			async function GDfetchGoogleElevations(latLngLocations) {
				await GDloadGoogleMapsApi();
				const elevations = [];
				for (let i = 0; i < latLngLocations.length; i += 512) elevations.push(...await GDfetchGoogleElevationsChunk(latLngLocations.slice(i, i + 512)));
				return elevations;
			}

			function GDbuildSampleLocationsForPoint(lat, lon) {
				const scales = [{ radius: 500, weight: 0.10, label: 'Local' }, { radius: 2000, weight: 0.30, label: 'Intermediate' }, { radius: 8000, weight: 0.60, label: 'Regional' }];
				const samples = [{ lat, lon, role: 'center', radius: 0 }];
				for (const scale of scales) { const cross = GDgenerateCrossPoints(lat, lon, scale.radius); samples.push({ ...cross.N, role: 'N', radius: scale.radius, scale }, { ...cross.E, role: 'E', radius: scale.radius, scale }, { ...cross.S, role: 'S', radius: scale.radius, scale }, { ...cross.W, role: 'W', radius: scale.radius, scale }); }
				return samples;
			}

			function GDcalculateFromElevations(lat, lon, elevations) {
				const scales = [{ radius: 500, weight: 0.10, label: 'Local' }, { radius: 2000, weight: 0.30, label: 'Intermediate' }, { radius: 8000, weight: 0.60, label: 'Regional' }];
				const zCenter = elevations[0];
				if (zCenter === null || zCenter === undefined || !isFinite(zCenter)) throw new Error('Could not retrieve center elevation.');
				let combinedGx = 0, combinedGy = 0, sinSum = 0, cosSum = 0, gradientSum = 0;
				const perRadius = [];
				for (let i = 0; i < scales.length; i++) {
					const scale = scales[i], base = 1 + (i * 4), zN = elevations[base], zE = elevations[base + 1], zS = elevations[base + 2], zW = elevations[base + 3];
					if ([zN, zE, zS, zW].some(z => z === null || z === undefined || !isFinite(z))) throw new Error(`Missing elevation data at ${scale.label} scale (${scale.radius} m).`);
					const g = GDfiniteDifferenceGradient(zN, zE, zS, zW, scale.radius);
					combinedGx += scale.weight * g.gx; combinedGy += scale.weight * g.gy;
					const dirRad = GDdegToRad(g.direction); sinSum += g.magnitude * Math.sin(dirRad); cosSum += g.magnitude * Math.cos(dirRad); gradientSum += g.magnitude;
					perRadius.push({ radius: scale.radius, label: scale.label, weight: scale.weight, gradient: g.magnitude, direction: g.direction, gx: g.gx, gy: g.gy });
				}
				const gradient = Math.sqrt(combinedGx * combinedGx + combinedGy * combinedGy), direction = (GDradToDeg(Math.atan2(-combinedGx, -combinedGy)) + 360) % 360;
				const R = gradientSum > 0 ? Math.sqrt(sinSum * sinSum + cosSum * cosSum) / gradientSum : 0;
				return { gradient, direction, centerElevation: zCenter, confidence: R > 0.85 ? 'high' : R > 0.60 ? 'moderate' : 'low', perRadius };
			}

			async function calculateGradientDirectionForBoxData() {
				if (!BoxData || !Array.isArray(BoxData.data) || BoxData.data.length === 0) throw new Error('BoxData.data is empty or unavailable.');
				await GDloadGoogleMapsApi();
				const pointRequests = [], allLocations = [];
				for (let i = 0; i < BoxData.data.length; i++) {
					const row = BoxData.data[i], lat = parseFloat(row[1]), lon = parseFloat(row[2]);
					if (!isFinite(lat) || !isFinite(lon)) continue;
					const start = allLocations.length, samples = GDbuildSampleLocationsForPoint(lat, lon);
					for (const sample of samples) allLocations.push(new google.maps.LatLng(sample.lat, sample.lon));
					pointRequests.push({ rowIndex: i, pointNumber: row[0], lat, lon, start, count: samples.length });
				}
				if (pointRequests.length === 0) throw new Error('No valid WGS84 point coordinates were found in BoxData.');
				const elevations = await GDfetchGoogleElevations(allLocations);
				GradientDirectionArray.length = 0;
				for (const request of pointRequests) {
					try { const result = GDcalculateFromElevations(request.lat, request.lon, elevations.slice(request.start, request.start + request.count)); GradientDirectionArray.push({ pointNumber: request.pointNumber, rowIndex: request.rowIndex, coordinate: { latitude: request.lat, longitude: request.lon }, gradient: result.gradient, direction: result.direction, centerElevation: result.centerElevation, confidence: result.confidence, perRadius: result.perRadius }); }
					catch (pointError) { GradientDirectionArray.push({ pointNumber: request.pointNumber, rowIndex: request.rowIndex, coordinate: { latitude: request.lat, longitude: request.lon }, gradient: null, direction: null, centerElevation: null, confidence: 'error', error: pointError.message }); }
				}
				window.GradientDirectionArray = GradientDirectionArray; window.BoxGradientDirectionData = GradientDirectionArray;
				GDappendGradientResultsToBoxData(); calculateAdvectiveHeatTransferForBoxData(); return GradientDirectionArray;
			}

			function GDformationOverlapThickness(formation, depthLimit) { const from = GDnumber(formation[0]), to = GDnumber(formation[1]); return Math.max(0, Math.min(to, depthLimit) - from); }
			function GDcalculateFormationFlowHeatValues(formation, gradient, direction, overlapThickness = null) {
				const from = GDnumber(formation[0]), to = GDnumber(formation[1]), fullThickness = Math.max(0, to - from), thickness = overlapThickness === null ? fullThickness : Math.max(0, overlapThickness);
				const matrixK = Math.max(0, GDnumber(formation[10])), porosityFraction = GDclamp(GDnumber(formation[11]) / 100, 0, 1), saturationFraction = GDclamp(GDnumber(formation[45]) / 100, 0, 1), effectivePorosityFraction = Math.max(porosityFraction * saturationFraction, 0);
				const darcyFlux = matrixK * Math.max(0, GDnumber(gradient)), groundwaterFlowRate = darcyFlux * thickness * saturationFraction, seepageVelocity = effectivePorosityFraction > 0 ? darcyFlux / effectivePorosityFraction : 0, advectiveHeatPotential = (groundwaterFlowRate / GD_SECONDS_PER_DAY) * GD_WATER_VOLUMETRIC_HEAT_CAPACITY;
				return { thickness, effectivePorosityPercent: effectivePorosityFraction * 100, darcyFlux, groundwaterFlowRate, seepageVelocity, advectiveHeatPotential, heatScore: advectiveHeatPotential, gradient: GDnumber(gradient), direction: GDnumber(direction) };
			}
			function GDappendGradientResultsToBoxData() {
				if (!BoxData || !Array.isArray(BoxData.data)) return;
				const byRowIndex = new Map(); GradientDirectionArray.forEach(item => byRowIndex.set(item.rowIndex, item));
				for (let i = 0; i < BoxData.data.length; i++) { const row = BoxData.data[i]; if (!row[3]) row[3] = []; GDensureArrayIndex(row[3], GD_POINT_DIRECTION_INDEX); const result = byRowIndex.get(i); row[3][GD_POINT_GRADIENT_INDEX] = result && Number.isFinite(result.gradient) ? GDroundValue(result.gradient, 8) : GDnumber(row[3][GD_POINT_GRADIENT_INDEX], 0); row[3][GD_POINT_DIRECTION_INDEX] = result && Number.isFinite(result.direction) ? GDroundValue(result.direction, 3) : GDnumber(row[3][GD_POINT_DIRECTION_INDEX], 0); }
			}
			function calculateAdvectiveHeatTransferForBoxData() {
				if (!BoxData || !Array.isArray(BoxData.data)) return [];
				const summary = []; let globalBest = { pointNumber: null, rowIndex: -1, formationNumber: 0, from: 0, to: 0, advectiveHeatPotential: -Infinity, heatScore: -Infinity };
				for (let i = 0; i < BoxData.data.length; i++) {
					const row = BoxData.data[i]; if (!row[3]) row[3] = []; GDensureArrayIndex(row[3], GD_POINT_BEST_SEEPAGE_VELOCITY_INDEX);
					const gradient = Math.max(0, GDnumber(row[3][GD_POINT_GRADIENT_INDEX])), direction = GDnumber(row[3][GD_POINT_DIRECTION_INDEX]), formations = Array.isArray(row[6]) ? row[6] : [];
					let best = { heatScore: -Infinity, from: 0, to: 0, formationNumber: 0, darcyFlux: 0, groundwaterFlowRate: 0, effectivePorosityPercent: 0, seepageVelocity: 0, advectiveHeatPotential: 0 };
					for (let j = 0; j < formations.length; j++) { const f = formations[j]; GDensureArrayIndex(f, GD_FORMATION_HEAT_SCORE_INDEX); const values = GDcalculateFormationFlowHeatValues(f, gradient, direction); f[GD_FORMATION_GRADIENT_INDEX] = GDroundValue(values.gradient, 8); f[GD_FORMATION_DIRECTION_INDEX] = GDroundValue(values.direction, 3); f[GD_FORMATION_DARCY_FLUX_INDEX] = GDroundValue(values.darcyFlux, 10); f[GD_FORMATION_GW_FLOW_INDEX] = GDroundValue(values.groundwaterFlowRate, 8); f[GD_FORMATION_EFFECTIVE_POROSITY_INDEX] = GDroundValue(values.effectivePorosityPercent, 6); f[GD_FORMATION_SEEPAGE_VELOCITY_INDEX] = GDroundValue(values.seepageVelocity, 8); f[GD_FORMATION_HEAT_POTENTIAL_INDEX] = GDroundValue(values.advectiveHeatPotential, 6); f[GD_FORMATION_HEAT_SCORE_INDEX] = GDroundValue(values.heatScore, 6); if (values.heatScore > best.heatScore) best = { heatScore: values.heatScore, from: GDnumber(f[0]), to: GDnumber(f[1]), formationNumber: j + 1, darcyFlux: values.darcyFlux, groundwaterFlowRate: values.groundwaterFlowRate, effectivePorosityPercent: values.effectivePorosityPercent, seepageVelocity: values.seepageVelocity, advectiveHeatPotential: values.advectiveHeatPotential }; }
					for (let d = 0; d < GD_BOREHOLE_DEPTHS.length; d++) { const depthLimit = GD_BOREHOLE_DEPTHS[d]; let cumulativePotential = 0; for (let j = 0; j < formations.length; j++) { const overlap = GDformationOverlapThickness(formations[j], depthLimit); if (overlap <= 0) continue; cumulativePotential += GDcalculateFormationFlowHeatValues(formations[j], gradient, direction, overlap).advectiveHeatPotential; } row[3][GD_POINT_DEPTH_POTENTIAL_START_INDEX + d] = GDroundValue(cumulativePotential, 6); }
					row[3][GD_POINT_BEST_HEAT_POTENTIAL_INDEX] = GDroundValue(Math.max(0, best.advectiveHeatPotential), 6); row[3][GD_POINT_BEST_FORMATION_FROM_INDEX] = GDroundValue(best.from, 3); row[3][GD_POINT_BEST_FORMATION_TO_INDEX] = GDroundValue(best.to, 3); row[3][GD_POINT_BEST_FORMATION_NUMBER_INDEX] = best.formationNumber; row[3][GD_POINT_BEST_DARCY_FLUX_INDEX] = GDroundValue(best.darcyFlux, 10); row[3][GD_POINT_BEST_GW_FLOW_INDEX] = GDroundValue(best.groundwaterFlowRate, 8); row[3][GD_POINT_BEST_EFFECTIVE_POROSITY_INDEX] = GDroundValue(best.effectivePorosityPercent, 6); row[3][GD_POINT_BEST_SEEPAGE_VELOCITY_INDEX] = GDroundValue(best.seepageVelocity, 8);
					if (best.heatScore > globalBest.heatScore) globalBest = { pointNumber: row[0], rowIndex: i, formationNumber: best.formationNumber, from: GDroundValue(best.from, 3), to: GDroundValue(best.to, 3), advectiveHeatPotential: GDroundValue(best.advectiveHeatPotential, 6), heatScore: GDroundValue(best.heatScore, 6), gradient: GDroundValue(gradient, 8), direction: GDroundValue(direction, 3) };
					summary.push({ pointNumber: row[0], gradient, direction, bestFormation: best, depthPotentials: GD_BOREHOLE_DEPTHS.map((depth, idx) => ({ depth, value: row[3][GD_POINT_DEPTH_POTENTIAL_START_INDEX + idx] })) });
				}
				window.AdvectiveHeatTransferArray = summary; window.BestAdvectiveHeatTransferLocation = globalBest; return summary;
			}
